=== Honeydew Grammar ===

-- NOTE: If viewing in kate, try setting syntax highlihgting to Lua, it works
--       best that way.

ExpressionLevel6:
	-- Primary expression
	NUMBER | STRING | keyword "true" | keyword "false" | keyword "null" | '(' Expression ')'

ExpressionLevel5:
	-- Remember that Level5 is recursive, since !(!(true)) can be an expression.
	( ('-' || '!') ExpressionLevel5 ) | ExpressionLevel6

ExpressionLevel4:
	ExpressionLevel5 ( ('*' || '/') ExpressionLevel5 ) ...

ExpressionLevel3:
	ExpressionLevel4 ( ('+' || '-') ExpressionLevel4 ) ...

ExpressionLevel2:
	ExpressionLevel3 ( ('>' || '>=' || '<' || '<=') ExpressionLevel3 ) ...

ExpressionLevel1:
	ExpressionLevel2 ( ('==' || '!=') ExpressionLevel2 ) ...

ExpressionLevel0:
	ExpressionLevel1 ( '?' ExpressionLevel0 ':' ExpressionLevel0 ) ...

Expression:
	ExpressionLevel0 ( ',' Expression ) ...

StatementLevel1:
	FunctionCall '(' Expression (',' Expression) ... ')'

StatementLevel0:
	StatementLevel1

Statement:
	( Expression || StatementLevel0 ) ';' ( Statement ) ...
